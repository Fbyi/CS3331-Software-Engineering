#include <fstream>

    string name;//物品名称
    string owner;//主人名称
    string phone;//联系方式
};
struct item arr[1000];//物品清单
int i = 0;//此时物品清单中物品个数

void Start(){
    ifstream fr;//读入
    ofstream fw;//输出
    fr.open("E:\\HelpEachOtherList.txt");//打开文件
    for(int j = 0; j < 1000; j++)
    {
        fr >> arr[j].name;
        fr >> arr[j].owner;
        fr >> arr[j].phone;
        if(arr[j].name != "\0") i += 1;
    }
    fr.close();//关闭文件
    ofstream outFile;
}

//Addition:执行添加操作，添加一个物品
void Addition(){
    cout << endl << "----------<添加>----------" <<endl;
    cout << "[请输入物品名称]:";
    string neo;
    cin >> neo;
    arr[i].name = neo;
    cout << "[请输入物品主人]:";
    cin >> neo;
    arr[i].owner = neo;
    cout << "[请输入联系电话]:";
    cin >> neo;
    arr[i].phone = neo;
    i = i + 1;
    cout << "-添加成功." << endl ;
    cout << "-------------------------" << endl;
}

//Deletion:执行删除功能，删除最后一件物品
void Deletion(){
    cout << endl << "----------<删除>----------" <<endl;
    cout << "[请输入需要删除的物品名称]:";
    string tar;//需要删除的物品名称
    cin >> tar;
    int SearchResult = 0;//是否有符合条件的物品
    int SeveralResult = 0;//符合条件的物品是否有多个
    for(int j=0; j < i; j++) {
        if (tar == arr[j].name) {
            SearchResult = 1;
            if(SeveralResult == 0)cout << "-这件物品符合要求." << endl;
            else{
                cout << endl << "-还有这件物品也符合要求." << endl;
            }
            cout << "物品名称" << "    " << "物品主人" << "    " << "联系方式" << endl;
            cout << arr[j].name << "    " << arr[j].owner << "    " << arr[j].phone << endl;
            cout << "1:<删除>" << endl;
            cout << "2:<不删除>" << endl;
            cout << "[你想要删除这件物品吗? 请输入1或2.]:";
            string del_order;//是否删除物品的指令
            int del;
            cin >> del_order;
            del = atoi(del_order.c_str());//防止非法输入
            if (del == 1) {
                for (int k = j; k < i; k++) {
                    arr[k].name = arr[k + 1].name;
                    arr[k].owner = arr[k + 1].owner;
                    arr[k].phone = arr[k + 1].phone;
                }
                i = i - 1;
                cout << "-删除成功." << endl;
            }
            if (del != 1) {
                cout << "-好的,没有删除." << endl;
            }
            SeveralResult = 1;
        }
    }
    if (SearchResult == 0){
        cout << "-清单中没有这项物品." << endl;
    }
    cout << "-------------------------" << endl;
}

//Check:执行查看功能，输出清单
void Check(){
    cout << endl << "----------<查看>----------" <<endl;
    if(i == 0){
        cout << "-清单中没有任何物品." << endl;
    }
    else {
        cout << "-清单中有以下物品:" << endl;
        cout << "物品名称" << "    " << "物品主人" << "    " << "联系方式" << endl;
        for(int j=0; j < i; j++){
            cout << arr[j].name << "    " << arr[j].owner << "    " << arr[j].phone << endl;
        }
    }
    cout << "-------------------------" << endl;
}

//Search:执行搜索功能，查找某一项物品并选择是否删除
void Search(){
    cout << endl << "----------<搜索>----------" <<endl;
    cout << "[请输入需要搜索的物品名称]:";
    string tar;//需要搜索的物品名称
    cin >> tar;
    int SearchResult = 0;//是否有符合条件的物品
    int TitleDone = 0;//是否已经输出了表格头
    for(int j=0; j < i; j++) {
        if (tar == arr[j].name) {
            SearchResult = 1;
            if (TitleDone == 0){
                TitleDone = 1;
                cout << "-以下物品符合要求:" << endl;
                cout << "物品名称" << "    " << "物品主人" << "    " << "联系方式" << endl;
            }
            cout << arr[j].name << "    " << arr[j].owner << "    " << arr[j].phone << endl;
        }
    }
    if(SearchResult == 0) {
        cout << "-清单中没有这项物品." << endl;
    }
    cout << "-------------------------" << endl;
}

//Error:给出非法输入提示

//Quit:将新的清单保存到清单文档中并退出程序
void Quit(){
    ifstream fr;//读入
    ofstream fw;//输出
    fw.open("E:\\HelpEachOtherList.txt");//打开文件
    //逐个写入物品
    for(int j = 0; j < i; j++)
    {
        fw << arr[j].name << endl;
        fw << arr[j].owner << endl;
        fw << arr[j].phone << endl;
    }
    //关闭文件
    fw.close();
    cout << endl << "----------<退出>----------" <<endl;
    cout << "-清单已保存，感谢使用." << endl;
    cout << "-------------------------" << endl << endl << endl;
    cout << "=======程序运行结束========";
}

int main() {
    Start();
    Readme();
    int over = 0;
    while(over == 0) {
        switch (Menu()) {
            case 1 :
                Addition();

            case 2 :
                Deletion();

            case 3 :
                Check();

            case 4 :

                break;
